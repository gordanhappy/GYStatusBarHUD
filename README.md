# GYStatusBarHUD
状态栏指示器

##成功提示
[GYStatusBarHUD showSuccess:@"加载成功!"];

##失败提示
 [GYStatusBarHUD showError:@"加载失败!"];

##加载中
[GYStatusBarHUD showLoading:@"正在加载中..."];

##文字和图片
[GYStatusBarHUD showMessage:@"文字" image:[UIImage imageNamed:@"check"]];

##文字提示
[GYStatusBarHUD showMessage:@"文字提示"];

##隐藏
[GYStatusBarHUD hide];
